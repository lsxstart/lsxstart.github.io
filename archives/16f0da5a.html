class Solution {
public:
    string replaceSpaces(string &str) {
        string res;
        for(char c : str){
            if(c == ' ') res += "%20";
            else res += c;
        }
        return res;
    }
};
class Solution {
public:
    //部分编程语言支持动态地将数组长度扩大，可以使用双指针算法，降低空间的使用
    string replaceSpaces(string &str) {
        int len;
        for(auto c : str){
            if(c == ' ') len += 3;
            else len ++;
        }
        
        int i = str.length() - 1,j = len - 1; //i定位到原数组的最后一位 j定位到新数组的最后一位
        
        str.resize(len); //重新扩容
        
        while(i >= 0){
            if(str[i] == ' '){
                str[j --] = '0';
                str[j --] = '2';
                str[j --] = '%';
            }else {
                str[j --] = str[i];
            }
            i--;
        }
        
        return str;
    }
};

短路操作终止递归

https://www.acwing.com/problem/content/80/

短路操作
  1. 什么是短路操作

在 C++ 中，递归是一种常见的编程技术，其中一个函数会调用自身来解决问题。然而，递归如果不正确处理，可能会导致栈溢出错误，特别是当递归深度非常大时。为了防止这种情况，通常需要一个终止条件，当满足这个条件时，递归调用将停止。

短路操作是一种在 C++ 中利用逻辑运算符的特性来终止递归的技术。C++ 中的逻辑运算符 &&（逻辑与）和 ||（逻辑或）支持短路评估。这意味着在表达式 expr1 && expr2 中，如果 expr1 为 false，那么 expr2 将不会被评估，因为整个表达式的结果已经确定为 false。类似地，在表达式 expr1 || expr2 中，如果 expr1 为 true，那么 expr2 将不会被评估，因为整个表达式的结果已经确定为 true。

利用短路操作终止递归的典型做法是将递归调用作为逻辑表达式的部分，并且确保当递归应该终止时，表达式的某个部分能够短路整个表达式。

class Solution {
public:
    int getSum(int n) {
        //递归、利用短路操作终止递归
        int res = n;
        (n > 0) && (res += getSum(n - 1));
        return res;
    }
};

单链表删除节点 O（1）

https://www.acwing.com/problem/content/85/

class Solution {
public:
    void deleteNode(ListNode* node) {
        //单链表 没有前驱节点 借下一个节点的数据 删除下一个节点
        ListNode* p = node -> next;
        
        node->val = p ->val;
        node->next = p -> next;
        
        delete p; //释放内存
    }
};

合并两个排序的链表（递归版本、迭代版本）

https://www.acwing.com/problem/content/submission/code_detail/35674016/

  1. 迭代版本

  1. 递归版本

递归时比较l1和l2的val，将更小的赋值给l1，最后返回l1即可。(多理解）

class Solution {
public:
    ListNode* merge(ListNode* l1, ListNode* l2) {
        //迭代版本
        ListNode* dummy = new ListNode(0);
        ListNode* cur = dummy;
        while(l1 != NULL && l2 != NULL){
            if(l1 -> val < l2 -> val){
                cur -> next = l1;
                l1 = l1 -> next;
            }else{
                cur -> next = l2;
                l2 = l2 -> next;
            }
            cur = cur -> next;
        }
        //接上剩余部分
        cur -> next = (l1 == NULL) ? l2 : l1;
        return dummy->next;
    }
};
class Solution {
public:
    ListNode* merge(ListNode* l1, ListNode* l2) {
        //处理l1或者l2为空的情况
        if(!l1 || !l2) return l1 ? l1 : l2;
        if(l1 -> val > l2 -> val) swap(l1,l2);
        l1 -> next = merge(l1 -> next ,l2);
        return l1;
    }
};

链表反转（递归版本、迭代版本）

https://www.acwing.com/problem/content/33/

  1. 递归版本

  1. 迭代版本

class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        //递归版本
        if(!head || ! ( head -> next) ) return head;
        //原来的尾节点就是反转后的头节点
        ListNode* tail = reverseList(head -> next);
        head -> next -> next = head;
        head -> next = nullptr;
        return tail;
    }
};
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        //迭代版本
        ListNode* cur = head;
        ListNode* pre = nullptr;
        ListNode* next; //保存节点
        while(cur != nullptr){
            next = cur -> next;
            cur -> next = pre;
            pre = cur,cur = next;
        }
        return pre;
    }
};

两个链表的第一个公共节点（ 普通方法、双指针法)

https://www.acwing.com/problem/content/description/62/

如果有公共结点肯定是在后面重叠，且后面部分都是共同的。

方法1：先计算出两个链表的长度，可以让比较长的先走两个链表长度之差的步数，两个再一起走。

方法2:

① 题解：

a. 用两个指针 p1，p2 分别指向两个链表 headA，headB 的头结点，同时向后遍历。

b. 当指针到达链表末尾时，重新定位到另一个链表的头结点。

c. 当它们相遇时，所指向的结点就是第一个公共结点。

② 解释：

class Solution {
public:
    ListNode *findFirstCommonNode(ListNode *headA, ListNode *headB) {
        //定义p永远指向较长的
        ListNode *p = headA,*q = headB;
        int lenA = 0,lenB = 0,k = 0;
        // for (auto t = headA; t; t = t->next) lenA ++;
        // for (auto t = headB; t; t = t->next) lenB ++;
        while(p){
            lenA ++;
            p = p->next;
        }
        while(q){
            lenB ++;
            q = q -> next;
        }
        p = headA,q = headB;
        k = lenA - lenB;
        if(lenB > lenA){ //保证p永远指向较长的链表
            p = headB,q = headA;
            k = lenB - lenA;
        }
        while(k--) p = p -> next;
        //目前p,q在同一起跑线
        while(p){
            if(p == q) return p;
            p = p -> next;
            q = q -> next;
        }
        //跳出 说明没有共同节点
        return nullptr;

    }
};
class Solution {
public:
    ListNode *findFirstCommonNode(ListNode *headA, ListNode *headB) {
        //双指针法寻找公共节点
        ListNode* p = headA;
        ListNode* q = headB;
        
        while(p != q){
            if(p) p = p -> next; //p不为空 移动到下一个节点
            else p = headB; //p为空 p移动到headB
            if(q) q = q -> next;
            else q = headA;
        }
        //至此，p,q相遇 为公共节点 即使没有公共节点 那也都是nullptr
        return p;
    }
};

把字符串转化成整数

https://www.acwing.com/problem/content/description/83/

class Solution {
public:
    int strToInt(string str) {
        int k = 0;
        while(k < str.size() && str[k] == ' ') k++;
        
        
        //普通int存放不下
        long long res = 0;
        
        //处理正负数
        int minus = 1;
        
        if(k < str.size()){
            if(str[k] == '-') minus = -1, k ++;
            else if(str[k] == '+') k++;
        }
        
        while(k < str.size() && str[k] >= '0' && str[k] <= '9'){
            res = res * 10 + str[k] - '0'; //这里的res 是绝对值
            if(res > 1e11) break;
            k++;
        }
        
        res *= minus; //处理负数
        
        if(res > INT_MAX) res = INT_MAX;
        if(res < INT_MIN) res = INT_MIN;
        
        
        return res;
         

    }
};

删除链表中重复的节点

https://www.acwing.com/problem/content/description/27/

  1. 首先考虑到边界问题，需设立一个虚拟头节点，统一操作
  2. 使用两个指针p,q；p用来遍历整个链表，q始终指向相同元素的最后一个位置
  3. p,q确定好后，如果q是p的下一个位置，说明没有重复元素，p移动到q的位置，否则就需要删除元素，即 p -> next = q -> next
class Solution {
public:
    ListNode* deleteDuplication(ListNode* head) {
        auto dummy = new ListNode(1); //创建虚拟头节点
        dummy -> next = head;
        
        auto p = dummy; //p来遍历链表
        while(p -> next){ 
            auto q = p -> next ;// q来寻找相同的元素
            while(q -> next && q -> val == q -> next -> val) q = q -> next; //q 始终指向重复元素的最后一个位置
            
            if(q == p -> next) p = q; //说明没有重复元素 p继续遍历
            else p -> next = q -> next; //有相同元素 删去
        }
        
        return dummy->next;
    }
};
